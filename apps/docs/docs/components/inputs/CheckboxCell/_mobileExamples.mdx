### Basic usage

```tsx
function CheckboxCellExample() {
  const [checked, setChecked] = React.useState(false);
  return (
    <CheckboxCell
      title="Enable notifications"
      description="Get updates on your account activity."
      checked={checked}
      onChange={(_value, newChecked) => setChecked(newChecked)}
      value="notifications"
    />
  );
}
```

### States

```tsx
<VStack gap={2}>
  <CheckboxCell
    title="Disabled and unchecked"
    value="disabled-unchecked"
    checked={false}
    disabled
  />
  <CheckboxCell title="Disabled and checked" value="disabled-checked" checked disabled />
</VStack>
```

### With Custom Content

```tsx
function CustomCheckboxCellExample() {
  const [checked, setChecked] = React.useState(true);
  return (
    <CheckboxCell
      checked={checked}
      description={
        <VStack gap={1}>
          <Text color="fgMuted" font="body">
            Stay updated with important information
          </Text>
          <Text color="fgPrimary" font="label1">
            • Security alerts
          </Text>
          <Text color="fgPrimary" font="label1">
            • Account updates
          </Text>
        </VStack>
      }
      onChange={(_value, newChecked) => setChecked(newChecked)}
      title={
        <HStack gap={2} alignItems="center">
          <RemoteImage source={assets.eth.imageUrl} size="m" />
          <VStack gap={0}>
            <Text font="headline">Email Notifications</Text>
            <Text color="fgPositive" font="caption">
              Recommended
            </Text>
          </VStack>
        </HStack>
      }
      value="notifications"
    />
  );
}
```
