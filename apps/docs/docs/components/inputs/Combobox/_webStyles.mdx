import { useState } from 'react';
import { ComponentStylesTable } from '@site/src/components/page/ComponentStylesTable';
import { StylesExplorer } from '@site/src/components/page/StylesExplorer';
import { Combobox } from '@coinbase/cds-web/alpha/combobox';

import webStylesData from ':docgen/web/alpha/combobox/Combobox/styles-data';

export const ComboboxExample = ({ classNames }) => {
  const [value, setValue] = useState();
  const options = [
    { value: 'option1', label: 'Option 1' },
    { value: 'option2', label: 'Option 2' },
    { value: 'option3', label: 'Option 3' },
  ];
  return (
    <Combobox
      classNames={classNames}
      label="Search options"
      options={options}
      placeholder="Type to search..."
      value={value}
      onChange={setValue}
      style={{ width: '100%' }}
    />
  );
};
